Throws

  • An exception if it fails

Implementation

double getTotalDown(int start, int end) {
  final OperationResult resultString = objectMethod(
    _pointerId,
    'RouteTerrainProfile',
    'getTotalDownBE',
    args: <String, int>{'first': start, 'second': end},
  );

  double result = resultString['result'];

  if (result.abs() < 0.0001) {
    result = 0;
  }

  return result;
}
