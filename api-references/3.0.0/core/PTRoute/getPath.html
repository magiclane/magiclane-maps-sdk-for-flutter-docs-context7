Path? getPath(final int start, final int end) {
  final OperationResult resultString = objectMethod(
    _pointerId,
    'RouteBase',
    'getPath',
    args: <String, int>{'start': start, 'end': end},
  );

  if (resultString['result'] == -1) {
    return null;
  }

  return Path.init(resultString['result']);
}
