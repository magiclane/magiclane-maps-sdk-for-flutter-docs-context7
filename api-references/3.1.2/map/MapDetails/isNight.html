static bool isNight(final Coordinates coords, final DateTime time) {
  final OperationResult resultString = staticMethod(
    'MapDetails',
    'isNight',
    args: <String, Object>{
      'coords': coords,
      'refTime': time.millisecondsSinceEpoch,
    },
  );

  return resultString['result'];
}
