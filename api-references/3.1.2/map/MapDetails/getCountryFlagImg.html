static Img? getCountryFlagImg(final String countryCode) {
  final OperationResult resultString = staticMethod(
    'MapDetails',
    'getCountryFlagImg',
    args: countryCode,
  );

  if (resultString['result'] == -1) {
    return null;
  }

  return Img.init(resultString['result']);
}
